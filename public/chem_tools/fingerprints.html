 99    def _get_atom_envs(self, mol: Chem.Mol) -> dict[int, list[str]]:
100        atoms_env: dict[int, list[str]] = {}
101        atom: Chem.Atom
102        for atom in mol.GetAtoms():  # type: ignore[call-arg, no-untyped-call]
103            idx = atom.GetIdx()
104            for radius in range(1, self.radius + 1):
105                if idx not in atoms_env:

111    def _find_env(cls, mol: Chem.Mol, idx: int, radius: int) -> str:
112        env = rdmolops.FindAtomEnvironmentOfRadiusN(mol, radius, idx)
113        atom_map: dict[int, int] = {}
114
115        submol = Chem.PathToSubmol(mol, env, atomMap=atom_map)
116        if idx in atom_map:

122    def _all_pairs(self, mol: Chem.Mol, atoms_env: dict[int, list[str]]) -> list[bytes]:
123        atom_pairs = []
124        distance_matrix = GetDistanceMatrix(mol)
125        num_atoms = mol.GetNumAtoms()
126        shingle_dict: defaultdict[str, int] = defaultdict(int)
127        for idx1, idx2 in itertools.combinations(range(num_atoms), 2):
128            dist = str(int(distance_matrix[idx1][idx2]))
129
130            for i in range(self.radius):
131                env_a = atoms_env[idx1][i]
132                env_b = atoms_env[idx2][i]
133
134                ordered = sorted([env_a, env_b])
135
136                shingle = f"{ordered[0]}|{dist}|{ordered[1]}"
137
138                if self.is_counted:
139                    shingle_dict[shingle] += 1
140                    shingle += "|" + str(shingle_dict[shingle])
141
142                atom_pairs.append(shingle.encode("utf-8"))
143        return list(set(atom_pairs))

def blockwise_array( arr_a: ~T, arr_b: ~T, sub_func: Callable[[~T, ~T], ~R], max_size: int = 1000, n_jobs: int = 1) -> ~R: 
21def blockwise_array(
22    arr_a: T,
23    arr_b: T,
24    sub_func: Callable[[T, T], R],
25    max_size: int = 1000,
26    n_jobs: int = 1,
27) -> R:
28    """Partwise applies a function to two arrays for a full matrix.
29
30    Examples:
31        >>> import numpy as np
32        >>> from clm_generator.python import blockwise_array
33        >>> def sub_func(a: np.ndarray, b: np.ndarray) -> np.ndarray:
34        ...     return a + b
35        >>> a = np.ones((200, 100))
36        >>> b = np.arange(300*100).reshape(300, 100)
37        >>> r = blockwise_array(a, b, sub_func, max_size=10)
38        >>> r.shape
39        (200, 300, 100)
40
41
42    Args:
43        arr_a: The first array or sequence.
44        arr_b: The second array or sequence.
45        sub_func: The function to apply to the subarrays.
46        max_size: The window size for the subarrays. Defaults to 1000.
47        n_jobs: The number of parallel jobs. Defaults to 1.
48
49    Returns:
50        The resulting array.
51    """
52    x_steps = [*list(range(0, len(arr_a), max_size)), len(arr_a)]
53    y_steps = [*list(range(0, len(arr_b), max_size)), len(arr_b)]
54
55    x_start_end = [(x_steps[ix], x_steps[ix + 1]) for ix in range(len(x_steps) - 1)]
56    y_start_end = [(y_steps[ix], y_steps[ix + 1]) for ix in range(len(y_steps) - 1)]
57
58    blocks: list[list[R]] = [[] for _ in range(len(x_start_end))]
59
60    data: list[tuple[T, T]] = []
61
62    for x_start, x_end in x_start_end:
63        for y_start, y_end in y_start_end:
64            sub_arr_a = arr_a[x_start:x_end]
65            sub_arr_b = arr_b[y_start:y_end]
66
67            data.append((sub_arr_a, sub_arr_b))
68
69    if n_jobs > 1:
70        result = auto_parallel(sub_func, data, n_jobs=n_jobs)
71    else:
72        result = [sub_func(sub_arr_a, sub_arr_b) for sub_arr_a, sub_arr_b in data]
73
74    for cell_ix, raw_sub_result in enumerate(result):
75        x_level = cell_ix // len(y_start_end)
76        if raw_sub_result.ndim == 3:  # noqa: PLR2004
77            sub_result = raw_sub_result.transpose((2, 0, 1))
78        else:
79            sub_result = raw_sub_result
80        blocks[x_level].append(sub_result)
81
82    combined: R = np.block(blocks)  # type: ignore[assignment]
83    if combined.ndim == 3:  # noqa: PLR2004
84        combined = combined.transpose(1, 2, 0)
85    return combined
