100    """Calculate the synthetic accessibility score of a molecule.
101
102    Args:
103        m: The molecule to calculate the synthetic accessibility score for.
104        fscores: A dictionary mapping fragment indices to their scores.
105            If None, the fragment scores are read from the RDKit contrib directory.
106        fp: The fingerprint of the molecule. If None, the fingerprint is calculated.
107
108    Returns:
109        The synthetic accessibility score of the molecule.
110    """
111    if fscores is None:
112        fscores = readFragmentScores()
113
114    if fp is None:
115        # fragment score
116        fp_gen = rdFPGen.GetMorganGenerator(
117            radius=2
118        )  # <- 2 is the *radius* of the circular fingerprint
119        fp = fp_gen.GetSparseCountFingerprint(m)
120
121    fps: dict[int, int] = fp.GetNonzeroElements()
122    score1 = 0.0
123    nf = 0
124    for bitId, v in fps.items():
125        nf += v
126        sfp = bitId
127        score1 += fscores.get(sfp, -4) * v
128    score1 /= nf
129
130    # features score
131    nAtoms = m.GetNumAtoms()
132    chiralCenters: list = Chem.FindMolChiralCenters(m, includeUnassigned=True)  # type: ignore[no-untyped-call]
133    nChiralCenters = len(chiralCenters)
134    ri = m.GetRingInfo()
135    nBridgeheads, nSpiro = numBridgeheadsAndSpiro(m, ri)
136    nMacrocycles = 0
137    for x in ri.AtomRings():
138        if len(x) > 8:
139            nMacrocycles += 1
140
141    sizePenalty = nAtoms**1.005 - nAtoms
142    stereoPenalty = math.log10(nChiralCenters + 1)
143    spiroPenalty = math.log10(nSpiro + 1)
144    bridgePenalty = math.log10(nBridgeheads + 1)
145    macrocyclePenalty = 0.0
146    # ---------------------------------------
147    # This differs from the paper, which defines:
148    #  macrocyclePenalty = math.log10(nMacrocycles+1) # noqa: ERA001
149    # This form generates better results when 2 or more macrocycles are present
150    if nMacrocycles > 0:
151        macrocyclePenalty = math.log10(2)
152
153    score2 = (
154        0.0
155        - sizePenalty
156        - stereoPenalty
157        - spiroPenalty
158        - bridgePenalty
159        - macrocyclePenalty
160    )
161
162    # correction for the fingerprint density
163    # not in the original publication, added in version 1.1
164    # to make highly symmetrical molecules easier to synthetise
165    score3 = 0.0
166    if nAtoms > len(fps):
167        score3 = math.log(float(nAtoms) / len(fps)) * 0.5
168
169    sascore = score1 + score2 + score3
170
171    # need to transform "raw" value into scale between 1 and 10
172    min = -4.0  # noqa: A001
173    max = 2.5  # noqa: A001
174    sascore = 11.0 - (sascore - min + 1) / (max - min) * 9.0
175    # smooth the 10-end
176    if sascore > 8.0:
177        sascore = 8.0 + math.log(sascore + 1.0 - 9.0)
178    if sascore > 10.0:
179        sascore = 10.0
180    elif sascore < 1.0:
181        sascore = 1.0
182
183    return sascore  # type: ignore[no-any-return]

207    # only process valid molecules

223
224        scores.append((ix, score))
